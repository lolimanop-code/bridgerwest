local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local VirtualInputManager = game:GetService("VirtualInputManager")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")
local LocalPlayer = Players.LocalPlayer

local MAX_BOBBER_DISTANCE = 40
local isFishing = false
local autoMode = false
local currentThread = nil
local isSpammingE = false
local spamConnection = nil

-- Переменные для сохранения позиции
local savedPosition = nil
local savedRotation = nil

-- Переменные для движения при испуганной рыбе
local moveDirection = "left"

-- Поток для отслеживания уведомлений
local spookedThread = nil
local spookedRunning = false

local chests = workspace.Chests

local MOVE_DISTANCE = 10
local TWEEN_TIME = 2

-- Настройки: укажи точное название удочки в игре
local FishingRodName = "FishingRod" -- <-- СЮДА ПИШИ НАЗВАНИЕ УДОЧКИ

-- Основная логика
local humanoid = nil
local backpack = nil

-- Функция обновления ссылок на живого персонажа
local function updateReferences()
    local character = LocalPlayer.Character
    if not character then return false end
    
    humanoid = character:FindFirstChild("Humanoid")
    backpack = LocalPlayer:FindFirstChild("Backpack")
    
    return humanoid ~= nil and backpack ~= nil
end

-- Функция экипировки
local function equipRod()
    if not humanoid or not backpack then
        if not updateReferences() then
            print("[⏳] Персонаж или инвентарь не загружены")
            return false
        end
    end
    
    local rod = backpack:FindFirstChild(FishingRodName)
    if rod then
        humanoid:EquipTool(rod)
		wait(0.3)
		humanoid:EquipTool(rod)
        print("[✅] Удочка '" .. FishingRodName .. "' в руке!")
        return true
    else
        print("[❌] Удочка '" .. FishingRodName .. "' не найдена в инвентаре")
        return false
    end
end

-- Когда персонаж появляется (в первый раз или после смерти) — обновляем ссылки
LocalPlayer.CharacterAdded:Connect(function(character)
    print("[🔄] Персонаж появился, обновляю ссылки...")
    character:WaitForChild("Humanoid")
    task.wait(0.1)
    updateReferences() 
end)

local function instantPrompts()
    for _, v in ipairs(chests:GetDescendants()) do
        if v:IsA("ProximityPrompt") then
            v.HoldDuration = 0
        end
    end
end

instantPrompts()

chests.DescendantAdded:Connect(function(desc)
    if desc:IsA("ProximityPrompt") then
        desc.HoldDuration = 0
    end
end)

-- Функция остановки спама E
local function stopSpammingE()
    if not isSpammingE then return end
    isSpammingE = false
    
    if spamConnection then
        spamConnection:Disconnect()
        spamConnection = nil
    end
    
    print("[СПАМ] Остановка спама клавиши E")
end

-- Функция сохранения позиции и поворота
local function savePlayerPosition()
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild('HumanoidRootPart') then
        local rootPart = LocalPlayer.Character.HumanoidRootPart
        savedPosition = rootPart.Position
        local _, yRotation, _ = rootPart.CFrame:ToOrientation()
        savedRotation = yRotation
        print("[СОХРАНЕНИЕ] Позиция сохранена:", savedPosition)
        print("[СОХРАНЕНИЕ] Поворот сохранён:", math.deg(savedRotation), "градусов")
        return true
    end
    return false
end

-- Функция телепортации на сохранённую позицию
local function teleportToSavedPosition()
    if not savedPosition or savedRotation == nil then
        print("[ОШИБКА] Нет сохранённой позиции или поворота!")
        return false
    end
    
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild('HumanoidRootPart') then
        local rootPart = LocalPlayer.Character.HumanoidRootPart
        local endPosition = CFrame.new(savedPosition.X,savedPosition.Y+2,savedPosition.Z) * CFrame.Angles(0, savedRotation, 0)
        local goal = {CFrame = endPosition}
        local tweenInfo = TweenInfo.new(TWEEN_TIME, Enum.EasingStyle.Quart)
        local tween = TweenService:Create(rootPart, tweenInfo, goal)
        tween:Play()
        tween.Completed:Wait()
        print("[ТЕЛЕПОРТАЦИЯ] Возврат на сохранённую позицию")
        return true
    end
    return false
end

local function movePlayer(direction)
    print("[ДВИЖЕНИЕ] Начинаю перемещение...")

    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild('HumanoidRootPart') then
        return false
    end
    
    local rootPart = LocalPlayer.Character.HumanoidRootPart
    local currentCFrame = rootPart.CFrame
    
    -- Получаем текущий поворот (чтобы сохранить его после перемещения)
    local _, yRotation, _ = currentCFrame:ToOrientation()
    
    -- Получаем вектор влево (отрицательный RightVector) или вправо
    local moveOffset
    if direction == "left" then
        local leftVector = -currentCFrame.RightVector
        moveOffset = leftVector * MOVE_DISTANCE
        print("[ДВИЖЕНИЕ] Перемещаюсь ВЛЕВО (относительно взгляда) на", MOVE_DISTANCE, "studs")
    else
        local rightVector = currentCFrame.RightVector
        moveOffset = rightVector * MOVE_DISTANCE
        print("[ДВИЖЕНИЕ] Перемещаюсь ВПРАВО (относительно взгляда) на", MOVE_DISTANCE, "studs")
    end
    
    -- Вычисляем новую позицию
    local newPosition = currentCFrame.Position + moveOffset
    
    -- Сохраняем поворот персонажа
    local endPosition = CFrame.new(newPosition) * CFrame.Angles(0, yRotation, 0)
    local goal = {CFrame = endPosition}
    
    local tweenInfo = TweenInfo.new(TWEEN_TIME, Enum.EasingStyle.Quart)
    local tween = TweenService:Create(rootPart, tweenInfo, goal)
    tween:Play()
    tween.Completed:Wait()
    
    print("[ДВИЖЕНИЕ] Перемещение завершено на позицию:", newPosition)
    return true
end

-- Функция остановки рыбалки
local function stopFishingActivity()
    if autoMode then
        print("[СТОП] Остановка рыбалки из-за испуганной рыбы")
        autoMode = false
        isFishing = false
        stopSpammingE()
    end
end

-- Функция обработки уведомления
local function handleNotification() 
    print("[HANDLE] Начало обработки уведомления")
    
    -- Останавливаем текущую рыбалку
    if autoMode then
        autoMode = false
        isFishing = false
        stopSpammingE()
    end
    
    task.wait(0.5)
    movePlayer(moveDirection)
    
    -- Меняем направление для следующего раза
    if moveDirection == "left" then
        moveDirection = "right"
    else
        moveDirection = "left"
    end
    
    task.wait(0.5)
    savePlayerPosition()
    
    -- Перезапускаем рыбалку
    print("[ЗАПУСК] Перезапуск рыбалки на новом месте...")
    startAutoFishing()
end

-- Функция остановки потока отслеживания
local function stopSpookedThread()
    spookedRunning = false
    -- Не закрываем корутину напрямую, просто даём ей завершиться самой
    spookedThread = nil
    print("[SPOOKED] Поток отслеживания остановлен")
end

-- Функция отслеживания испуганной рыбы
local function spooked()
    print("[SPOOKED] Ожидание появления NotifierGui...")
    
    -- Ждём появления NotifierGui
    while not LocalPlayer.PlayerGui:FindFirstChild("NotifierGui") and spookedRunning do
        task.wait(1)
    end
    
    if not spookedRunning then return end
    
    local notifierGui = LocalPlayer.PlayerGui:FindFirstChild("NotifierGui")
    if not notifierGui then return end
    
    print("[SPOOKED] NotifierGui найден!")
    
    -- Ждём появления MessageContainer
    local messageContainer = nil
    while not messageContainer and spookedRunning do
        messageContainer = notifierGui:FindFirstChild("MessageContainer")
        if not messageContainer then
            task.wait(0.5)
        end
    end
    
    if not spookedRunning then return end
    print("[SPOOKED] MessageContainer найден!")

    local function findTargetNotification()
        for _, child in ipairs(messageContainer:GetChildren()) do
            if child.Name == "Notification" or child.Name == "notification" then
                local need = "The fish here are spooked. Move to a new spot!"
                if child.Text == need then
                    print("[SPOOKED] Найдено уведомление об испуганной рыбе!")
                    return true
                end
            end
        end
        return false
    end

    print("[SPOOKED] Начинаю бесконечный цикл проверки уведомлений...")
    
    while spookedRunning do
        if findTargetNotification() then
            print("[SPOOKED] Обнаружено уведомление! Вызываю handleNotification()")
            handleNotification()
            -- После обработки уведомления поток продолжает работать
            -- Нужно подождать, пока уведомление скроется
            task.wait(2)
        end
        task.wait(0.5)
    end
    
    print("[SPOOKED] Поток отслеживания завершён")
end

-- Функция запуска потока отслеживания
local function startSpookedThread()
    -- Останавливаем старый поток, если есть
    if spookedThread then
        stopSpookedThread()
    end
    
    -- Запускаем новый поток
    spookedRunning = true
    spookedThread = task.spawn(function()
        spooked()
    end)
    print("[SPOOKED] Поток отслеживания запущен")
end

-- Функция клика в центр экрана
local function clickCenterScreen()
    local viewportSize = workspace.CurrentCamera.ViewportSize
    local centerX = viewportSize.X / 2
    local centerY = viewportSize.Y / 2
    
    VirtualInputManager:SendMouseButtonEvent(centerX, centerY-(centerY/2), 0, true, game, 1)
    task.wait(0.05)
    VirtualInputManager:SendMouseButtonEvent(centerX, centerY-(centerY/2), 0, false, game, 1)
    
    print("[КЛИК] Нажатие в центр экрана")
end

-- Функция эмуляции нажатия клавиши
local function pressKey(keyCode)
    if type(keyCode) == "string" then
        local keyMap = {
            ["A"] = Enum.KeyCode.A, ["B"] = Enum.KeyCode.B, ["C"] = Enum.KeyCode.C,
            ["D"] = Enum.KeyCode.D, ["E"] = Enum.KeyCode.E, ["F"] = Enum.KeyCode.F,
            ["G"] = Enum.KeyCode.G, ["H"] = Enum.KeyCode.H, ["I"] = Enum.KeyCode.I,
            ["J"] = Enum.KeyCode.J, ["K"] = Enum.KeyCode.K, ["L"] = Enum.KeyCode.L,
            ["M"] = Enum.KeyCode.M, ["N"] = Enum.KeyCode.N, ["O"] = Enum.KeyCode.O,
            ["P"] = Enum.KeyCode.P, ["Q"] = Enum.KeyCode.Q, ["R"] = Enum.KeyCode.R,
            ["S"] = Enum.KeyCode.S, ["T"] = Enum.KeyCode.T, ["U"] = Enum.KeyCode.U,
            ["V"] = Enum.KeyCode.V, ["W"] = Enum.KeyCode.W, ["X"] = Enum.KeyCode.X,
            ["Y"] = Enum.KeyCode.Y, ["Z"] = Enum.KeyCode.Z,
        }
        keyCode = keyMap[string.upper(keyCode)]
        if not keyCode then return end
    end
    
    VirtualInputManager:SendKeyEvent(true, keyCode, false, game)
    task.wait(0.02)
    VirtualInputManager:SendKeyEvent(false, keyCode, false, game)
end

-- Функции спама E
local function startSpammingE()
    if isSpammingE then return end
    isSpammingE = true
    
    print("[СПАМ] Запуск спама клавиши E")
    
    spamConnection = RunService.Heartbeat:Connect(function()
        if isSpammingE and autoMode then
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.E, false, game)
            task.wait(0.01)
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.E, false, game)
        end
    end)
end

-- Функция обработки QTE
local function handleQTE()
    local mashingSystem = LocalPlayer.PlayerGui:FindFirstChild("MashingSystem")
    if not mashingSystem then return false end
    
    local container = mashingSystem:FindFirstChild("Container")
    if not container then return false end
    
    local waitTime = 0
    
    while waitTime < 1 and autoMode do
        if container.Visible == true then
            print("[QTE] Окно появилось!")
            break
        end
        task.wait(0.1)
        waitTime = waitTime + 0.1
    end
    
    if not container or container.Visible ~= true then
        return false
    end
    
    stopSpammingE()
    
    local circle = container:FindFirstChild("Circle")
	local lastText = ""
	local lastChangeTime = tick()
	local timeoutSeconds = 2  -- Если 1.5 секунды нет новых клавиш — QTE закончен

	while autoMode and container.Visible == true do
		if circle then
			local keyLabel = circle:FindFirstChild("KeyLabel")
			if keyLabel then
				local requiredKey = keyLabel.Text
				if requiredKey and requiredKey ~= "" then
					
					-- Если текст изменился — обновляем таймер
					if requiredKey ~= lastText then
						lastText = requiredKey
						lastChangeTime = tick()
						print("[QTE] Новая клавиша:", requiredKey)
					end
					
					-- Проверяем, не прошло ли слишком много времени без новых клавиш
					if tick() - lastChangeTime > timeoutSeconds then
						print("[QTE] Таймаут: новых клавиш нет", timeoutSeconds, "секунд")
						break  -- Выходим из цикла
					end
					
					pressKey(requiredKey)
					task.wait(0.05)
				end
			end
		end
		task.wait(0.05)
	end

	-- Теперь телепорт сработает в любом случае
	teleportToSavedPosition()
    if container.Visible == false then
        print("[QTE] Успешно завершено!")
        return true
    end
    
    return false
end

-- Поиск ближайшего поплавка
local function getClosestBobber()
    local character = LocalPlayer.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then
        return nil
    end
    
    local rootPart = character.HumanoidRootPart
    local playerPos = rootPart.Position
    
    local closestBobber = nil
    local closestDistance = MAX_BOBBER_DISTANCE + 1
    
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj:IsA("Part") or obj:IsA("MeshPart") then
            if not obj:IsDescendantOf(character) then
                local distance = (obj.Position - playerPos).Magnitude
                if distance <= MAX_BOBBER_DISTANCE and distance < closestDistance then
                    closestDistance = distance
                    closestBobber = obj
                end
            end
        end
    end
    
    return closestBobber, closestDistance
end

-- Поиск WaterSplashContainer
local function findSplashForBobber(bobber)
    if not bobber then return nil end
    
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj.Name == "WaterSplashContainer" then
            local distance = (obj.Position - bobber.Position).Magnitude
            if distance < 20 then
                return obj, distance
            end
        end
    end
    
    return nil
end

-- Отслеживание поклёвки
local function waitForBite(bobber)
    if not bobber then return false end
    
    print("[⏳] Ожидание появления WaterSplashContainer...")
    
    local splash = nil
    local waitTime = 0
    
    while waitTime < 20 and autoMode do
        splash = findSplashForBobber(bobber)
        if splash then
            print("[🟢] WaterSplashContainer появился!")
            break
        end
        task.wait(0.5)
        waitTime = waitTime + 0.5
    end
    
    if not splash or not autoMode then
        print("[❌] WaterSplashContainer не появился")
        return false
    end
    
    print("[🎯] Отслеживаю приближение сплеша...")
    
    while autoMode and splash and splash.Parent do
        local distance = (splash.Position - bobber.Position).Magnitude
        
        if distance < 0.5 then
            print("[🐟] Сплеш у поплавка! Клюёт!")
            clickCenterScreen()
            task.wait(0.1)
            clickCenterScreen()
			task.wait(0.1)
            clickCenterScreen()
            return true
        end
        
        task.wait(0.05)
    end
    
    return false
end

-- Один цикл рыбалки
local function doOneFishingCycle()
    if not autoMode then return false end
    
    isFishing = true
    print("[🎣] Заброс удочки...")
    
    clickCenterScreen()
    
    print("[⏳] Ожидание поплавка...")
    task.wait(2)
    
    local bobber, distance = getClosestBobber()
    
    if not bobber then
        print("[❌] Поплавок не найден!")
        isFishing = false
        return false
    end
    
    print("========================================")
    print("🎣 ПОПЛАВОК НАЙДЕН!")
    print("📛 Название:", bobber.Name)
    print("📏 Расстояние:", string.format("%.1f", distance), "studs")
    print("========================================")
    
    startSpammingE()
    
    local biteSuccess = waitForBite(bobber)
    
    stopSpammingE()
    
    if not biteSuccess then
        isFishing = false
        print("[⚠️] Поклёвки не произошло")
        return false
    end
    
    print("[QTE] Проверка наличия QTE...")
    task.wait(0.5)
    
    local qteSuccess = handleQTE()
    
    if qteSuccess then
        print("[✅] Рыба поймана и QTE пройдено!")
    else
        print("[⚠️] QTE не появилось или не пройдено")
    end
    
    isFishing = false
    return true
end

-- Запуск авторежима
function startAutoFishing()
    if autoMode then
        print("[⚠️] Авторежим уже запущен")
        return
    end
    
    savePlayerPosition()
    
    -- Запускаем поток отслеживания (если ещё не запущен)
    if not spookedThread then
        startSpookedThread()
    end
    
    autoMode = true
    print("========================================")
    print("🤖 АВТОРЫБАЛКА ЗАПУЩЕНА")
    print("📍 Сохранена позиция для возврата")
    print("📢 Отслеживаются уведомления об испуганной рыбе")
    print("↔️ При испуге: движение влево/вправо на", MOVE_DISTANCE, "studs")
    print("⌨️ Нажмите K для остановки")
    print("========================================")
    
    currentThread = task.spawn(function()
        while autoMode do
            local success = doOneFishingCycle()
            
            if success and autoMode then
                print("Новый заброс")
            elseif autoMode then
                print("[⏸️] Повтор через 1 секунду...")
                task.wait(1)
            end
        end
    end)
end

-- Остановка авторежима
local function stopAutoFishing()
    if not autoMode then
        print("[ℹ️] Авторежим не был запущен")
        return
    end
    
    print("[🛑] Останавливаю авторежим...")
    autoMode = false
    isFishing = false
    stopSpammingE()
    
    -- Останавливаем поток отслеживания
    stopSpookedThread()
    
    task.wait(0.5)
end

-- Обработка клавиши K
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.K then
		equipRod()
        if autoMode then
            stopAutoFishing()
        else
            startAutoFishing()
        end
    end
end)
