local MAX_BOBBER_DISTANCE = 60
local isFishing = false
local autoMode = false
local currentThread = nil
local isSpammingE = false

local savedPosition = nil
local savedRotation = nil

local moveDirection = "left"

local spookedThread = nil

local chests = workspace.Chests

local TWEEN_TIME = 2

local FishingRodName = "FishingRod"

local humanoid = nil
local backpack = nil

local isEquipping = false
local rodMonitorConnection = nil
local periodicThread = nil

local characterAddedConnection = nil
local chestsDescendantConnection = nil

local function updateReferences()
    local character = LocalPlayer.Character
    if not character then return false end
    
    humanoid = character:FindFirstChild("Humanoid")
    backpack = LocalPlayer:FindFirstChild("Backpack")
    
    return humanoid ~= nil and backpack ~= nil
end

local function isRodInHands()
    local character = LocalPlayer.Character
    if not character then return false end
    return character:FindFirstChild(FishingRodName) ~= nil
end

local function forceEquipRod()
    if isEquipping then return false end
    if not updateReferences() then return false end
    
    if isRodInHands() then return true end
    
    isEquipping = true
    local rod = backpack:FindFirstChild(FishingRodName)
    if rod then
        humanoid:EquipTool(rod)
        task.wait(0.2)
        humanoid:EquipTool(rod)
        print("[✅] Удочка экипирована!")
        isEquipping = false
        return true
    end
    isEquipping = false
    return false
end

local function stopRodControl()
    if rodMonitorConnection then
        rodMonitorConnection:Disconnect()
        rodMonitorConnection = nil
    end
    if periodicThread then
        task.cancel(periodicThread)
        periodicThread = nil
    end
    print("[КОНТРОЛЬ] Остановка контроля удочки")
end

local function startRodControl()
    stopRodControl()
    
    if not autoMode then return end
    
    local character = LocalPlayer.Character
    if not character then return end
    
    print("[КОНТРОЛЬ] Запуск контроля удочки")
    
    rodMonitorConnection = character.ChildRemoved:Connect(function(child)
        if child:IsA("Tool") and child.Name == FishingRodName then
            print("[⚠️] Удочка убрана из рук! Возвращаю...")
            task.wait(0.1)
            forceEquipRod()
        end
    end)
end

instantPrompts()

local function savePlayerPosition()
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild('HumanoidRootPart') then
        local rootPart = LocalPlayer.Character.HumanoidRootPart
        savedPosition = rootPart.Position
        local _, yRotation, _ = rootPart.CFrame:ToOrientation()
        savedRotation = yRotation
        print("[СОХРАНЕНИЕ] Позиция сохранена:", savedPosition)
        print("[СОХРАНЕНИЕ] Поворот сохранён:", math.deg(savedRotation), "градусов")
        return true
    end
    return false
end

local function isNearSavedPosition(maxDistance)
    if not savedPosition then return false end
    
    local character = LocalPlayer.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then
        return false
    end
    
    local rootPart = character.HumanoidRootPart
    local distance = (rootPart.Position - savedPosition).Magnitude
    
    return distance <= maxDistance
end

local function forceTeleportToSavedPosition()
    if not savedPosition or savedRotation == nil then
        print("[ОШИБКА] Нет сохранённой позиции или поворота!")
        return false
    end
    
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild('HumanoidRootPart') then
        return false
    end
    
    local rootPart = LocalPlayer.Character.HumanoidRootPart
    local endPosition = CFrame.new(savedPosition.X, savedPosition.Y, savedPosition.Z) * CFrame.Angles(0, savedRotation, 0)
    local goal = {CFrame = endPosition}
    local tweenInfo = TweenInfo.new(TWEEN_TIME, Enum.EasingStyle.Quart)
    local tween = TweenService:Create(rootPart, tweenInfo, goal)
    tween:Play()
    tween.Completed:Wait()
    
    rootPart.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
    rootPart.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    
    local actualDistance = (rootPart.Position - savedPosition).Magnitude
    print("[ТЕЛЕПОРТ] Фактическое расстояние до цели:", string.format("%.1f", actualDistance), "studs")
    
    return true
end

local function teleportToSavedPosition()
    if not savedPosition or savedRotation == nil then
        print("[ОШИБКА] Нет сохранённой позиции или поворота!")
        return false
    end
    
    local maxAttempts = 3
    local acceptableDistance = 5
    
    for attempt = 1, maxAttempts do
        print("[ТЕЛЕПОРТ] Попытка", attempt, "из", maxAttempts)
        
        forceTeleportToSavedPosition()
        
        task.wait(0.05)
        
        if isNearSavedPosition(acceptableDistance) then
            print("[ТЕЛЕПОРТ] ✅ Успешно! Игрок на позиции (попытка", attempt, ")")
            return true
        else
            local character = LocalPlayer.Character
            if character and character:FindFirstChild("HumanoidRootPart") then
                local currentPos = character.HumanoidRootPart.Position
                local diff = (currentPos - savedPosition).Magnitude
                print("[ТЕЛЕПОРТ] ❌ Промах! Расстояние:", string.format("%.1f", diff), "studs")
            end
            
            if attempt < maxAttempts then
                print("[ТЕЛЕПОРТ] 🔄 Пробую снова...")
                task.wait(0.15)
            end
        end
    end
    
    print("[ТЕЛЕПОРТ] ⚠️ Не удалось телепортироваться после", maxAttempts, "попыток")
    return false
end

local function movePlayer(direction)
    print("[ДВИЖЕНИЕ] Начинаю перемещение...")

    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild('HumanoidRootPart') then
        return false
    end
    
    local rootPart = LocalPlayer.Character.HumanoidRootPart
    local currentCFrame = rootPart.CFrame
    local currentY = currentCFrame.Position.Y
    local _, yRotation, _ = currentCFrame:ToOrientation()
    
    local moveOffset
    if direction == "left" then
        moveOffset = -currentCFrame.RightVector * MOVE_DISTANCE
        print("[ДВИЖЕНИЕ] Перемещаюсь ВЛЕВО на", MOVE_DISTANCE, "studs")
    else
        moveOffset = currentCFrame.RightVector * MOVE_DISTANCE
        print("[ДВИЖЕНИЕ] Перемещаюсь ВПРАВО на", MOVE_DISTANCE, "studs")
    end
    
    local newPos = currentCFrame.Position + moveOffset
    local endPosition = CFrame.new(newPos.X, currentY, newPos.Z) * CFrame.Angles(0, yRotation, 0)
    local goal = {CFrame = endPosition}
    
    local tweenInfo = TweenInfo.new(TWEEN_TIME, Enum.EasingStyle.Quart)
    local tween = TweenService:Create(rootPart, tweenInfo, goal)
    tween:Play()
    tween.Completed:Wait()
    
    rootPart.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
    rootPart.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    
    print("[ДВИЖЕНИЕ] Перемещение завершено. Y остался:", currentY)
    return true
end

local function stopFishingActivity()
    if autoMode then
        print("[СТОП] Остановка рыбалки из-за испуганной рыбы")
        autoMode = false
        isFishing = false
        stopSpammingE()
    end
end

local function handleNotification()
    print("[HANDLE] Начало обработки уведомления")
    
    if autoMode then
        autoMode = false
        isFishing = false
        stopSpammingE()
    end
    
    task.wait(0.5)
    movePlayer(moveDirection)
    
    if moveDirection == "left" then
        moveDirection = "right"
    else
        moveDirection = "left"
    end
    
    task.wait(0.5)
    savePlayerPosition()
    
    print("[ЗАПУСК] Перезапуск рыбалки на новом месте...")
    startAutoFishing()
end

local function stopSpookedThread()
    spookedRunning = false
    if spookedThread then
        task.cancel(spookedThread)
        spookedThread = nil
    end
    print("[SPOOKED] Поток отслеживания остановлен")
end

local function spooked()
    while spookedRunning do
        local notifierGui = LocalPlayer.PlayerGui:FindFirstChild("NotifierGui")
        if notifierGui then
            local messageContainer = notifierGui:FindFirstChild("MessageContainer")
            if messageContainer then
                local found = false
                for _, child in ipairs(messageContainer:GetChildren()) do
                    if child.Name == "Notification" or child.Name == "notification" then
                        if child.Text == "The fish here are spooked. Move to a new spot!" then
                            print("[SPOOKED] Найдено уведомление об испуганной рыбе!")
                            handleNotification()
                            found = true
                            break
                        end
                    end
                end
                if found then
                    task.wait(2)
                end
            end
        end
        task.wait(1) -- было 0.5, но логика та же, просто реже проверка
    end
end

local function startSpookedThread()
    stopSpookedThread()
    
    spookedRunning = true
    spookedThread = task.spawn(function()
        spooked()
    end)
    print("[SPOOKED] Поток отслеживания запущен")
end

local function clickCenterScreen()

    local centerX = viewportSize.X / 2
    local centerY = viewportSize.Y / 2
    
    VirtualInputManager:SendMouseButtonEvent(centerX, centerY - (centerY / 2), 0, true, game, 1)
    task.wait(0.01)
    VirtualInputManager:SendMouseButtonEvent(centerX, centerY - (centerY / 2), 0, false, game, 1)
    
    print("[КЛИК] Нажатие в центр экрана")
end

local function pressKey(keyCode)
    if type(keyCode) == "string" then
        local keyMap = {
            ["A"] = Enum.KeyCode.A, ["B"] = Enum.KeyCode.B, ["C"] = Enum.KeyCode.C,
            ["D"] = Enum.KeyCode.D, ["E"] = Enum.KeyCode.E, ["F"] = Enum.KeyCode.F,
            ["G"] = Enum.KeyCode.G, ["H"] = Enum.KeyCode.H, ["I"] = Enum.KeyCode.I,
            ["J"] = Enum.KeyCode.J, ["K"] = Enum.KeyCode.K, ["L"] = Enum.KeyCode.L,
            ["M"] = Enum.KeyCode.M, ["N"] = Enum.KeyCode.N, ["O"] = Enum.KeyCode.O,
            ["P"] = Enum.KeyCode.P, ["Q"] = Enum.KeyCode.Q, ["R"] = Enum.KeyCode.R,
            ["S"] = Enum.KeyCode.S, ["T"] = Enum.KeyCode.T, ["U"] = Enum.KeyCode.U,
            ["V"] = Enum.KeyCode.V, ["W"] = Enum.KeyCode.W, ["X"] = Enum.KeyCode.X,
            ["Y"] = Enum.KeyCode.Y, ["Z"] = Enum.KeyCode.Z,
        }
        keyCode = keyMap[string.upper(keyCode)]
        if not keyCode then return end
    end
    
    VirtualInputManager:SendKeyEvent(true, keyCode, false, game)
    task.wait(0.02)
    VirtualInputManager:SendKeyEvent(false, keyCode, false, game)
end

local function handleQTE()
    local mashingSystem = LocalPlayer.PlayerGui:FindFirstChild("MashingSystem")
    if not mashingSystem then return false end
    
    local container = mashingSystem:FindFirstChild("Container")
    if not container then return false end
    
    local waitTime = 0
    
    while waitTime < 1 and autoMode do
        if container.Visible == true then
            print("[QTE] Окно появилось!")
            break
        end
        task.wait(0.1)
        waitTime = waitTime + 0.1
    end
    
    if not container or container.Visible ~= true then
        return false
    end
    
    stopSpammingE()
    
    local circle = container:FindFirstChild("Circle")
    local lastText = ""
    local lastChangeTime = tick()
    local timeoutSeconds = 2

    while autoMode and container.Visible == true do
        if circle then
            local keyLabel = circle:FindFirstChild("KeyLabel")
            if keyLabel then
                local requiredKey = keyLabel.Text
                if requiredKey and requiredKey ~= "" then
                    if requiredKey ~= lastText then
                        lastText = requiredKey
                        lastChangeTime = tick()
                        print("[QTE] Новая клавиша:", requiredKey)
                    end
                    
                    if tick() - lastChangeTime > timeoutSeconds then
                        print("[QTE] Таймаут: новых клавиш нет", timeoutSeconds, "секунд")
                        break
                    end
                    
                    pressKey(requiredKey)
                    task.wait(0.05)
                end
            end
        end
        task.wait(0.05)
    end

    teleportToSavedPosition()
    if container.Visible == false then
        print("[QTE] Успешно завершено!")
        return true
    end
    
    return false
end

local function getClosestBobber()
    local character = LocalPlayer.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then
        return nil
    end
    
    local rootPart = character.HumanoidRootPart
    local playerPos = rootPart.Position
    
    local closestBobber = nil
    local closestDistance = MAX_BOBBER_DISTANCE + 1
    
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj:IsA("Part") or obj:IsA("MeshPart") then
            if not obj:IsDescendantOf(character) then
                local distance = (obj.Position - playerPos).Magnitude
                if distance <= MAX_BOBBER_DISTANCE and distance < closestDistance then
                    closestDistance = distance
                    closestBobber = obj
                end
            end
        end
    end
    
    return closestBobber, closestDistance
end

local function findSplashForBobber(bobber)
    if not bobber then return nil end
    
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj.Name == "WaterSplashContainer" then
            local distance = (obj.Position - bobber.Position).Magnitude
            if distance < 20 then
                return obj, distance
            end
        end
    end
    
    return nil
end

local function waitForBite(bobber)
    if not bobber then return false end
    
    print("[⏳] Ожидание появления WaterSplashContainer...")
    
    local splash = nil
    local waitTime = 0
    
    while waitTime < 20 and autoMode do
        splash = findSplashForBobber(bobber)
        if splash then
            print("[🟢] WaterSplashContainer появился!")
            break
        end
        task.wait(0.5)
        waitTime = waitTime + 0.5
    end
    
    if not splash or not autoMode then
        print("[❌] WaterSplashContainer не появился")
        return false
    end
    
    print("[🎯] Отслеживаю приближение сплеша...")
    
    while autoMode and splash and splash.Parent do
        local distance = (splash.Position - bobber.Position).Magnitude
        
        if distance < 0.6 then
            print("[🐟] Сплеш у поплавка! Клюёт!")
            clickCenterScreen()
            task.wait(0.05)
            clickCenterScreen()
            task.wait(0.05)
            clickCenterScreen()
            return true
        end
        
        task.wait(0.05) -- было 0.02, но всё ещё быстро и без утечки
    end
    
    return false
end

local function doOneFishingCycle()
    if not autoMode then return false end
    
    if not isRodInHands() then
        print("[⚠️] Удочка не в руках перед циклом! Экипирую...")
        forceEquipRod()
        task.wait(0.3)
    end
    
    isFishing = true
    print("[🎣] Заброс удочки...")
    
    local existingBobber = getClosestBobber()
    if existingBobber then
        print("[⚠️] Поплавок уже в воде! Пропускаю заброс...")
    else
        clickCenterScreen()
        task.wait(0.5)
    end
    
    print("[⏳] Ожидание поплавка...")
    
    local bobber = nil
    local distance = nil
    local waitTime = 0
    local maxWaitTime = 3.5
    
    while waitTime < maxWaitTime and autoMode do
        bobber, distance = getClosestBobber()
        if bobber then
            print("[✅] Поплавок появился через", string.format("%.1f", waitTime), "сек")
            break
        end
        task.wait(0.2)
        waitTime = waitTime + 0.2
    end
    
    if not bobber then
        print("[❌] Поплавок не появился! Удочка не сработала.")
        print("[🔄] Переэкипирую удочку и пробую снова...")
        
        local character = LocalPlayer.Character
        if character then
            local currentTool = character:FindFirstChildOfClass("Tool")
            if currentTool then
                humanoid:UnequipTools()
                task.wait(0.2)
            end
        end
        
        forceEquipRod()
        task.wait(0.3)
        
        isFishing = false
        return false
    end
    
    print("========================================")
    print("🎣 ПОПЛАВОК НАЙДЕН!")
    print("📛 Название:", bobber.Name)
    print("📏 Расстояние:", string.format("%.1f", distance), "studs")
    print("========================================")
    
    startSpammingE()
    
    local biteSuccess = waitForBite(bobber)
    
    stopSpammingE()
    
    if not biteSuccess then
        isFishing = false
        print("[⚠️] Поклёвки не произошло")
        task.wait(0.5)
        return false
    end
    
    print("[QTE] Проверка наличия QTE...")
    task.wait(0.3)
    
    local qteSuccess = handleQTE()
    
    if qteSuccess then
        print("[✅] Рыба поймана и QTE пройдено!")
    else
        print("[⚠️] QTE не появилось или не пройдено")
    end
    
    task.wait(0.3)
    
    isFishing = false
    return true
end

function startAutoFishing()
    if autoMode then
        print("[⚠️] Авторежим уже запущен")
        return
    end
    
    savePlayerPosition()
    
    if not spookedThread then
        startSpookedThread()
    end
    
    autoMode = true
    print("========================================")
    print("🤖 АВТОРЫБАЛКА ЗАПУЩЕНА")
    print("========================================")
    
    forceEquipRod()
    startRodControl()
    
    currentThread = task.spawn(function()
        while autoMode do
            local success = doOneFishingCycle()
            if success and autoMode then
                print("Новый заброс")
            elseif autoMode then
                task.wait(1)
            end
        end
    end)
end

local function stopAutoFishing()
    if not autoMode then
        print("[ℹ️] Авторежим не был запущен")
        return
    end
    
    print("[🛑] Останавливаю авторежим...")
    autoMode = false
    isFishing = false
    
    stopSpammingE()
    stopSpookedThread()
    stopRodControl()
    
    if currentThread then
        task.cancel(currentThread)
        currentThread = nil
    end
    
    task.wait(0.5)
end

        -- Получаем remote и NPC заново каждый раз (на случай перезагрузки)
        local remotes = game:GetService("ReplicatedStorage"):FindFirstChild("Remotes")
        local sellRemote = remotes and remotes:FindFirstChild("DialogueRemote")

        local daniel = npcFolder and npcFolder:FindFirstChild("Daniel")
        
        if sellRemote and daniel then
            local ignoredTools = {FishingRod=true, AmmoPack=true, Stand=true}
            local soldCount = 0
            
            -- Продажа из рюкзака
            if LP.Backpack then
                local toolsToSell = {}
                for _, tool in ipairs(LP.Backpack:GetChildren()) do
                    if tool:IsA("Tool") and not ignoredTools[tool.Name] and not tool:FindFirstChild("AmmoInClip") then
                        table.insert(toolsToSell, tool)
                    end
                end
                
                for _, tool in ipairs(toolsToSell) do
                    pcall(function() 
                        sellRemote:FireServer("Action", "SellAll_" .. tool.Name, daniel)
                        soldCount = soldCount + 1
                    end)
                    task.wait(0.05)
                end
            end
            
            -- Продажа из рук
            local c = getChar()
            if c then
                local toolsToSell = {}
                for _, tool in ipairs(c:GetChildren()) do
                    if tool:IsA("Tool") and not ignoredTools[tool.Name] and not tool:FindFirstChild("AmmoInClip") then
                        table.insert(toolsToSell, tool)
                    end
                end
                
                for _, tool in ipairs(toolsToSell) do
                    pcall(function() 
                        sellRemote:FireServer("Action", "SellAll_" .. tool.Name, daniel)
                        soldCount = soldCount + 1
                    end)
                    task.wait(0.05)
                end
            end
            
            if soldCount > 0 then
                print("[AUTO SELL] Продано предметов: " .. soldCount)
            else
                print("[AUTO SELL] Нет предметов для продажи")
            end
        else
            print("[AUTO SELL] Remote или NPC не найдены")
        end
        
        task.wait(CHECK_INTERVAL)
    end
