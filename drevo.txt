local VirtualInputManager = game:GetService("VirtualInputManager")

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

local UserInputService = game:GetService("UserInputService")


local function killPlayer()
    local character = LocalPlayer.Character  -- Берём АКТУАЛЬНОГО персонажа
    if character then
        local humanoid = character:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.Health = 0
        end
    end
end

local function tpplayer(spot)
	killPlayer()
	local newCharacter = LocalPlayer.CharacterAdded:Wait()
	newCharacter:WaitForChild("Humanoid")
	wait(1)
	if spot == "farm" then
		LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(-1478.921875, 78.54988861083984, -3867.71826171875)
	elseif spot == "roka" then
		LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(-8044.02783203125, 67.37040710449219, -2285.8623046875)
	end

end

local function waitForKey(keyCode)
    print("Ожидание нажатия: " .. tostring(keyCode))
    
    local connection = nil
    local pressed = false
    
    connection = UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        if input.KeyCode == keyCode then
            print("✅ Нажата клавиша: " .. tostring(keyCode))
            pressed = true
        end
    end)
    
    repeat task.wait() until pressed
    connection:Disconnect()
end

local function SimulateMouseClick(x, y)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 1)
    wait(0.05)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 1)
end

local tween_s = game:GetService('TweenService')

local lp = game.Players.LocalPlayer

function bypass_teleport(v)
    if lp.Character and 
    lp.Character:FindFirstChild('HumanoidRootPart') then
		local distance = (lp.Character.HumanoidRootPart.Position - v).Magnitude
		local sec = distance/35
		local tweeninfo = TweenInfo.new(sec,Enum.EasingStyle.Linear)
        local cf = CFrame.new(v)
        local a = tween_s:Create(lp.Character.HumanoidRootPart,tweeninfo,{CFrame=cf})
        a:Play()
        a.Completed:Wait()
        -- print('Teleporting Done!')
    end
end

local UserInputService = game:GetService("UserInputService")

local isRunning = false
local loopCoroutine = nil

local TOGGLE_KEY = Enum.KeyCode.L

local function stopLoop()
    if isRunning then
        isRunning = false
    end
    
    if loopCoroutine then
        coroutine.close(loopCoroutine)
        loopCoroutine = nil
    end
end


local function startLoop()
    if isRunning then
        return  
    end
    
    isRunning = true
    loopCoroutine = coroutine.create(function()
        while isRunning do
			for i, v in pairs(teleport_table) do
				bypass_teleport(v)
				for k = 1, (15/0.1) do
					SimulateMouseClick(500, 500)
					wait(0.1)
				end
			end
        end
    end)
    
    coroutine.resume(loopCoroutine)
end


local function toggleLoop()
    if isRunning then
        stopLoop()
    else
		tpplayer("farm")
        startLoop()
    end
end

-- Настройки: укажи точное название удочки в игре
local FishingRodName = "LumberAxe" -- <-- СЮДА ПИШИ НАЗВАНИЕ УДОЧКИ

local humanoid = nil
local backpack = nil

-- Функция обновления ссылок на живого персонажа
local function updateReferences()
    local character = LocalPlayer.Character
    if not character then return false end
    
    humanoid = character:FindFirstChild("Humanoid")
    backpack = LocalPlayer:FindFirstChild("Backpack")
    
    return humanoid ~= nil and backpack ~= nil
end

updateReferences()

local ITEM_NAME = "Rokakaka Seed"  -- Название предмета для отслеживания

-- Функция, которая выполнится когда предмет появится
local function onItemAdded(item)
    if item.Name == ITEM_NAME then
        print("[НАЙДЕН] Предмет '" .. ITEM_NAME .. "' появился в инвентаре!")
        
        -- 👇 ТВОЙ КОД ЗДЕСЬ (что должно произойти)

        tpplayer("roka")
		waitForKey(Enum.KeyCode.T)
    end
end

-- Следим за появлением новых предметов
local processed = false

local itemMonitorRunning = false
local itemMonitorThread = nil
local processed = false

-- ============ ФУНКЦИЯ ОСТАНОВКИ МОНИТОРИНГА ============
local function stopItemMonitor()
    itemMonitorRunning = false
    if itemMonitorThread then
        task.cancel(itemMonitorThread)
        itemMonitorThread = nil
    end
    processed = false
    print("[МОНИТОР] Остановлен")
end

local function equipAxe()
    if not humanoid or not backpack then
        if not updateReferences() then
            print("[⏳] Персонаж или инвентарь не загружены")
            return false
        end
    end
	updateReferences()
	wait(0.5)
    local rod = backpack:FindFirstChild(FishingRodName)
    if rod then
        humanoid:EquipTool(rod)
		wait(0.3)
		humanoid:EquipTool(rod)
		print("[✅] Удочка '" .. FishingRodName .. "' в руке!")
        return true
    else
        print("[❌] Удочка '" .. FishingRodName .. "' не найдена в инвентаре")
        return false
    end
end

-- ============ ФУНКЦИЯ ЗАПУСКА МОНИТОРИНГА ============
local function startItemMonitor()
    if itemMonitorRunning then
        print("[МОНИТОР] Уже запущен")
        return
    end
    
    itemMonitorRunning = true
    itemMonitorThread = task.spawn(function()
        while itemMonitorRunning do
			task.wait(0.3)
			
			if not processed then
				local backpack = LocalPlayer:FindFirstChild("Backpack")
				if backpack then
					local item = backpack:FindFirstChild(ITEM_NAME)
					if item then
						processed = true
						print("[НАЙДЕН] Предмет '" .. ITEM_NAME .. "' появился в инвентаре!")
						
						toggleLoop()
						tpplayer("roka")
						waitForKey(Enum.KeyCode.T)
						toggleLoop()
						equipAxe()
						task.wait(1)
						processed = false
					end
				end
			end
		end
    end)
    print("[МОНИТОР] Запущен")
end

-- ============ ПЕРЕКЛЮЧЕНИЕ МОНИТОРИНГА ============
local function toggleItemMonitor()
    if itemMonitorRunning then
        stopItemMonitor()
    else
        startItemMonitor()
    end
end

-- Проверяем уже существующие предметы
if backpack then
    for _, item in ipairs(backpack:GetChildren()) do
        onItemAdded(item)
    end
end

-- Когда персонаж появляется (в первый раз или после смерти) — обновляем ссылки
LocalPlayer.CharacterAdded:Connect(function(character)
    print("[🔄] Персонаж появился, обновляю ссылки...")
    character:WaitForChild("Humanoid")
    task.wait(0.1)
    updateReferences()
end)

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == TOGGLE_KEY then
		toggleItemMonitor()
        toggleLoop()
		equipAxe()
    end
end)
