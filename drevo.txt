local function waitForKey(keyCode)
    print("Ожидание нажатия: " .. tostring(keyCode))
    
    local connection = nil
    local pressed = false
    
    connection = UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        if input.KeyCode == keyCode then
            print("✅ Нажата клавиша: " .. tostring(keyCode))
            pressed = true
        end
    end)
    
    repeat task.wait() until pressed
    connection:Disconnect()
end

local function SimulateMouseClick(x, y)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 1)
    wait(0.05)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 1)
end

local isRunning = false
local loopCoroutine = nil

local TOGGLE_KEY = Enum.KeyCode.K

local function stopLoop()
    if isRunning then
        isRunning = false
    end
    
    if loopCoroutine then
        coroutine.close(loopCoroutine)
        loopCoroutine = nil
    end
end


local function startLoop()
    if isRunning then
        return  
    end
    
    isRunning = true
    loopCoroutine = coroutine.create(function()
        while isRunning do
			for i, v in pairs(teleport_table) do
				bypass_teleport(v)
				for k = 1, (15/0.1) do
					SimulateMouseClick(500, 500)
					wait(0.1)
				end
			end
        end
    end)
    
    coroutine.resume(loopCoroutine)
end


local function toggleLoop()
    if isRunning then
        stopLoop()
    else
		tpplayer(farm)
        startLoop()
    end
end

-- Настройки: укажи точное название удочки в игре
local FishingRodName = "LumberAxe" -- <-- СЮДА ПИШИ НАЗВАНИЕ УДОЧКИ

local humanoid = nil
local backpack = nil

-- Функция обновления ссылок на живого персонажа
local function updateReferences()
    local character = LocalPlayer.Character
    if not character then return false end
    
    humanoid = character:FindFirstChild("Humanoid")
    backpack = LocalPlayer:FindFirstChild("Backpack")
    
    return humanoid ~= nil and backpack ~= nil
end

-- Функция экипировки
local function equipAxe()
    if not humanoid or not backpack then
        if not updateReferences() then
            print("[⏳] Персонаж или инвентарь не загружены")
            return false
        end
    end
    
    local rod = backpack:FindFirstChild(FishingRodName)
    if rod then
        humanoid:EquipTool(rod)
		wait(0.3)
		humanoid:EquipTool(rod)
		print("[✅] Удочка '" .. FishingRodName .. "' в руке!")
        return true
    else
        print("[❌] Удочка '" .. FishingRodName .. "' не найдена в инвентаре")
        return false
    end
end

-- Когда персонаж появляется (в первый раз или после смерти) — обновляем ссылки
LocalPlayer.CharacterAdded:Connect(function(character)
    print("[🔄] Персонаж появился, обновляю ссылки...")
    character:WaitForChild("Humanoid")
    task.wait(0.1)
    updateReferences()
end)

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == TOGGLE_KEY then
        toggleLoop()
		equipAxe()
    end
end)
